                <button :style="`background-color:${color}`" >
    <!-- @slot text of the button -->
    <slot />
                </button>
              </template>

              <script>
              export default {
                props: {
    /**
                   * Set the background color of the button
                   * @values blue, red, green
                   */
    color: {
                    type: String,
                    default: "blue"
                  }
                }
              }
              </script>
